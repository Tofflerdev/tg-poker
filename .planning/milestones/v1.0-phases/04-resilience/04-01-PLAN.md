```ts
// Step 1: capture currentChips
const user = await prisma.user.findUnique({
  where: { telegramId: BigInt(Number(telegramId)) },
  select: { currentChips: true }
});
if (!user || user.currentChips === null) return null;

const chipsToRefund = user.currentChips;

// Step 2: atomic refund + clear, idempotency guard
const result = await prisma.user.updateMany({
  where: { telegramId: BigInt(Number(telegramId)), currentChips: { not: null } },
  data:  {
    balance: { increment: chipsToRefund },
    currentChips: null,
    currentTableId: null,
    currentSeat: null,
    disconnectedAt: null,
    lastSeenAt: null
  }
});

if (result.count === 0) return null; // race: another caller already cleared
return { refunded: chipsToRefund };
```

ANTI-PATTERNS (DO NOT USE):
- `prisma.$queryRaw` — D-D1 explicitly chose `updateMany`
- `prisma.$transaction(async (tx) => { read; write })` — race window in default isolation; D-D1 chose single-statement atomicity
- Inverting order (atomic first, then read) — would lose the chips amount
- Removing the `IS NOT NULL` guard from step 2 — opens double-credit on race

```ts
   *
```

```ts
   * caller sees `count === 0` and returns null — no double write.
```
