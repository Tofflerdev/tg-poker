- Renders desktop layout when `useIsMobile()` returns false. The mobile layout has a separate "Bet Panel" sub-view (showBetPanel state) — for happy-path tests, target the DESKTOP layout to avoid the slide-up panel state.
- Action emits flow through `emitAction(action, ...args)` which calls `(socket.emit as any)(action, ...args)`.
- Fold: `emitAction("fold")` — emits 'fold' with no args.
- Call/Check: clicked when toCall > 0 → `emitAction("call")`; toCall === 0 → `emitAction("check")`.
- Raise: desktop Raise button → `emitAction("raise", raiseAmount)` directly.
- All-In: `emitAction("allIn")` (capital I).
- Component returns a "Hand Complete / Waiting" status panel when `gameState.stage === 'waiting' || 'showdown'`.
- Component returns "X is thinking..." panel when `mySeat === null || mySeat !== currentPlayer || folded || allIn`.
- For tests to render the action bar: `gameState.stage` must be `'preflop' | 'flop' | 'turn' | 'river'` AND `gameState.currentPlayer === mySeat` AND seat must not be folded/allIn.

    import { render, fireEvent } from '@testing-library/react';

    // Motion passthrough — copy of ReconnectOverlay.test.tsx pattern. Empty seats
    // don't render motion components, but occupied-seat tests render HandDisplay
    // which may transitively use motion in future. Defensive mock.
    vi.mock('motion/react', async () => {
      const ReactMod = await import('react');
      const passthrough = (tag: string) =>
        ReactMod.forwardRef<HTMLElement, any>((props, ref) => {
          const { initial, animate, exit, transition, variants, whileHover, whileTap, layout, ...rest } = props;
          return ReactMod.createElement(tag, { ...rest, ref });
        });
      const motion: any = new Proxy({}, { get: (_t, tag: string) => passthrough(tag) });
      return {
        motion,
        AnimatePresence: ({ children }: any) => ReactMod.createElement(ReactMod.Fragment, null, children),
        useReducedMotion: () => false,
      };
    });

    function makePlayer(overrides: Partial<Player> = {}): Player {

        id: 'u-occupied',
