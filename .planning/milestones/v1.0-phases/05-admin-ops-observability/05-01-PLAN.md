```typescript
import type { Socket } from 'socket.io';
import type {
  ExtendedClientEvents,
  ExtendedServerEvents,
  SocketData,
  TelegramUser,
} from '../../types/index.js';
import type { DefaultEventsMap } from 'socket.io';

/**
 * Phase 5 / Plan 05-01 (COMPLIANCE-04 / D-13 / D-14 / RESEARCH Open Q3).
 *
 * Server-side gate for joinTable. Returns true if the user may proceed to the
 * existing balance + seat flow; returns false and emits a typed `serverError`
 * to the caller's socket otherwise.
 *
 * Order: ban first (banned users never see the Consent screen), ToS second.
 *
 * The user object may be either the in-memory TelegramUser (server/models/User.ts)
 * or a Prisma User row — both carry `tosAcceptedAt` and `bannedAt`. We accept any
 * shape with those two fields so callers don't need to reshape.
 */
export interface JoinGateUser {
  tosAcceptedAt?: string | Date | null;
  bannedAt?: string | Date | null;
}

type GateSocket = Socket<
  ExtendedClientEvents,
  ExtendedServerEvents,
  DefaultEventsMap,
  SocketData
>;

export function gateUserOrEmit(user: JoinGateUser, socket: GateSocket): boolean {
  // D-RESEARCH-Q3: ban check first. A banned user is rejected even when ToS is missing.
  if (user.bannedAt) {
    socket.emit('serverError', { type: 'BANNED' });
    return false;
  }
  // D-13 / D-14: ALL users with no ToS acceptance are gated — no createdAt cutoff.
  if (!user.tosAcceptedAt) {
    socket.emit('serverError', { type: 'TOS_REQUIRED' });
    return false;
  }
  return true;
}
```

Note: the `JoinGateUser` interface accepts string|Date|null so the helper works with both the in-memory TelegramUser (string) and a fresh Prisma row (Date). The truthiness check `if (user.bannedAt)` correctly handles both `null`/`undefined` and treats any non-null Date as banned.

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-5-01-1 | Elevation of privilege | Banned user reseating after ban via repeated joinTable | mitigate | `gateUserOrEmit` in joinTable handler reads in-memory `userStorage.getUser(telegramId).bannedAt`; mirror is refreshed by Plan 05-04 admin ban handler immediately after the DB write so the next `joinTable` returns BANNED. |
| T-5-01-2 | Bypass of consent | New user (no `tosAcceptedAt`) seating before tapping Accept | mitigate | Server checks `user.tosAcceptedAt` before any seat / balance action. Failure emits `serverError TOS_REQUIRED`; client is informational — server is the authority. |
| T-5-01-3 | Information disclosure | `serverError` payload leaking PII | accept | The payload is a closed string union with no PII; `{ type: 'TOS_REQUIRED' | 'BANNED' }` only. |
| T-5-01-4 | Tampering | Client invents a fake `tosAcceptedAt` to bypass gate | mitigate | Server reads `tosAcceptedAt` from `userStorage` (which is populated from DB during auth via `mapToTelegramUser`). Client cannot write back. The existing `acceptTos` handler is the only write path and validates `payload.version`. |
| T-5-01-5 | Race | User accepts ToS in tab A, joinTable in tab B before in-memory cache updates | accept | The existing `acceptTos` handler (Plan 02-08) mirrors `result.tosAcceptedAt` into `user.tosAcceptedAt` synchronously after the DB write. Window is sub-millisecond per process. Cross-process state is irrelevant in MVP single-instance deployment. |
